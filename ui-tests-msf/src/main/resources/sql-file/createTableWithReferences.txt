DROP TABLE IF EXISTS artist;
CREATE TABLE artist (
    cp__artistid VARCHAR(128) PRIMARY KEY,
    l__artistname__artistnickname VARCHAR(128),
    a__artistname VARCHAR(128),
    a__artisttitle VARCHAR(128)
);

DROP TABLE IF EXISTS track;
CREATE TABLE track (
    cp__trackid VARCHAR(128),
    a__trackname VARCHAR(128), 
    r__artist VARCHAR(128),
    f__number NUMERIC(12,2)
);
