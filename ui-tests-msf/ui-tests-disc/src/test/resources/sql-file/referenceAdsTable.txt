DROP TABLE IF EXISTS artist;
CREATE TABLE artist (
    cp__artistid VARCHAR(128) PRIMARY KEY,
    l__artistname__artistnickname VARCHAR(128),
    a__artistname VARCHAR(128),
    a__artisttitle VARCHAR(128)
);

DROP TABLE IF EXISTS track;
CREATE TABLE track (
    cp__trackid VARCHAR(128),
    a__trackname VARCHAR(128), 
    r__artist VARCHAR(128),
    f__number NUMERIC(12,2)
); 

TRUNCATE TABLE artist;
TRUNCATE TABLE track;
COPY artist(cp__artistid, l__artistname__artistnickname, a__artistname, a__artisttitle) FROM LOCAL '${DATA_SOURCE_DIR}/artist.csv' DELIMITER ',' SKIP 1 ABORT ON ERROR;
COPY track(cp__trackid, a__trackname, r__artist, f__number) FROM LOCAL '${DATA_SOURCE_DIR}/track.csv' DELIMITER ',' SKIP 1 ABORT ON ERROR;
